## Answer to the Question 1

#include <stdio.h>
#include <math.h>

int main() {
    float energy, mass, speed_of_light;

    printf("Enter energy and mass: ");
    scanf("%f %f", &energy, &mass);
    
    speed_of_light = sqrt(energy / mass);

    printf("Speed of light: %.3f\n", speed_of_light);

    return 0;
}
